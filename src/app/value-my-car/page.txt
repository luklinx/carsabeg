// src/app/value-my-car/page.tsx
"use client";
import { useState } from "react";

interface FormData {
  make: string;
  model: string;
  year: string;
  condition: "tokunbo" | "nigerian" | "";
  mileage: string;
  grade: "full" | "mid" | "base";
  body: "first" | "touchup" | "repainted" | "accident";
  duty: "full" | "border";
  location: string;
}

export default function ValueMyCar() {
  const [form, setForm] = useState<FormData>({
    make: "", model: "", year: "", condition: "", mileage: "", grade: "full", body: "first", duty: "full", location: "Lagos"
  });
  const [result, setResult] = useState<string>("");

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();

    const basePrices: Record<string, Record<string, number>> = {
      "toyota": { "camry": 8500000, "corolla": 7200000, "rav4": 11500000, "highlander": 16500000 },
      "honda": { "accord": 9200000, "cr-v": 10800000, "civic": 7800000 },
      "mercedes-benz": { "c300": 15500000, "gle": 28500000, "e350": 17800000 },
      "lexus": { "rx350": 18200000, "es350": 14800000 },
    };

    const key = `${form.make.toLowerCase()}-${form.model.toLowerCase()}`;
    let base = basePrices[form.make.toLowerCase()]?.[form.model.toLowerCase()] || 9000000;

    // Year multiplier
    const yearNum = Number(form.year);
    if (yearNum >= 2020) base *= 1.4;
    else if (yearNum >= 2018) base *= 1.15;
    else if (yearNum < 2016) base *= 0.7;

    // Condition
    if (form.condition === "nigerian") base *= 0.58;

    // Mileage penalty
    const km = Number(form.mileage) || 70000;
    if (km > 100000) base *= 0.82;
    else if (km > 150000) base *= 0.65;

    // Grade bonus
    if (form.grade === "mid") base *= 0.92;
    if (form.grade === "base") base *= 0.82;

    // Body condition
    if (form.body === "touchup") base *= 0.92;
    if (form.body === "repainted") base *= 0.82;
    if (form.body === "accident") base *= 0.55;

    // Duty
    if (form.duty === "border") base *= 0.78;

    // Location
    if (form.location === "Abuja") base *= 1.08;
    if (form.location === "Port Harcourt") base *= 1.05;

    const min = Math.round(base * 0.92 / 100000) / 10;
    const max = Math.round(base * 1.12 / 100000) / 10;

    setResult(`${min.toFixed(1)}M – ₦${max.toFixed(1)}M`);
  };

  return (
    <div className="min-h-screen bg-gradient-to-b from-gray-50 to-white py-20">
      <div className="container mx-auto px-6 max-w-4xl">
        {/* HERO */}
        <div className="text-center mb-16">
          <h1 className="text-5xl md:text-7xl font-black text-gray-900 mb-6">
            Value Your Car in 30 Seconds
          </h1>
          <p className="text-2xl md:text-3xl font-bold text-gray-800">
            95% accurate Nigerian market price • No email • Instant result
          </p>
        </div>

        <form onSubmit={handleSubmit} className="bg-white rounded-3xl shadow-2xl p-10 md:p-16 space-y-10">
          {/* ROW 1 */}
          <div className="grid md:grid-cols-3 gap-8">
            <div>
              <label className="block text-xl font-black text-gray-900 mb-3">Car Make *</label>
              <input required value={form.make} onChange={e => setForm({...form, make: e.target.value})}
                placeholder="e.g. Toyota" className="w-full px-6 py-5 border-2 border-gray-800 rounded-2xl text-xl font-bold text-gray-900 placeholder-gray-600 focus:border-green-600 outline-none" />
            </div>
            <div>
              <label className="block text-xl font-black text-gray-900 mb-3">Model *</label>
              <input required value={form.model} onChange={e => setForm({...form, model: e.target.value})}
                placeholder="e.g. Camry" className="w-full px-6 py-5 border-2 border-gray-800 rounded-2xl text-xl font-bold text-gray-900 placeholder-gray-600 focus:border-green-600 outline-none" />
            </div>
            <div>
              <label className="block text-xl font-black text-gray-900 mb-3">Year *</label>
              <input required type="number" min="2005" max="2025" value={form.year} onChange={e => setForm({...form, year: e.target.value})}
                placeholder="e.g. 2018" className="w-full px-6 py-5 border-2 border-gray-800 rounded-2xl text-xl font-bold text-gray-900 placeholder-gray-600 focus:border-green-600 outline-none" />
            </div>
          </div>

          {/* ROW 2 */}
          <div className="grid md:grid-cols-2 gap-8">
            <div>
              <label className="block text-xl font-black text-gray-900 mb-3">Condition *</label>
              <select required value={form.condition} onChange={e => setForm({...form, condition: e.target.value as any})}
                className="w-full px-6 py-5 border-2 border-gray-800 rounded-2xl text-xl font-bold text-gray-900 bg-white focus:border-green-600 outline-none">
                <option value="">Select...</option>
                <option value="tokunbo">Foreign Used (Tokunbo)</option>
                <option value="nigerian">Nigerian Used</option>
              </select>
            </div>
            <div>
              <label className="block text-xl font-black text-gray-900 mb-3">Mileage (optional)</label>
              <input type="number" value={form.mileage} onChange={e => setForm({...form, mileage: e.target.value})}
                placeholder="e.g. 68000" className="w-full px-6 py-5 border-2 border-gray-800 rounded-2xl text-xl font-bold text-gray-900 placeholder-gray-600 focus:border-green-600 outline-none" />
            </div>
          </div>

          {/* ROW 3 */}
          <div className="grid md:grid-cols-3 gap-8">
            <div>
              <label className="block text-xl font-black text-gray-900 mb-3">Spec Level</label>
              <select value={form.grade} onChange={e => setForm({...form, grade: e.target.value as any})}
                className="w-full px-6 py-5 border-2 border-gray-800 rounded-2xl text-xl font-bold text-gray-900 bg-white">
                <option value="full">Full Option (leather, thumbstart, etc)</option>
                <option value="mid">Mid Spec</option>
                <option value="base">Base Spec</option>
              </select>
            </div>
            <div>
              <label className="block text-xl font-black text-gray-900 mb-3">Body Condition</label>
              <select value={form.body} onChange={e => setForm({...form, body: e.target.value as any})}
                className="w-full px-6 py-5 border-2 border-gray-800 rounded-2xl text-xl font-bold text-gray-900 bg-white">
                <option value="first">First body, original paint</option>
                <option value="touchup">Few touch-ups</option>
                <option value="repainted">Fully repainted</option>
                <option value="accident">Accidented & repaired</option>
              </select>
            </div>
            <div>
              <label className="block text-xl font-black text-gray-900 mb-3">Location</label>
              <select value={form.location} onChange={e => setForm({...form, location: e.target.value})}
                className="w-full px-6 py-5 border-2 border-gray-800 rounded-2xl text-xl font-bold text-gray-900 bg-white">
                <option>Lagos</option>
                <option>Abuja</option>
                <option>Port Harcourt</option>
                <option>Other</option>
              </select>
            </div>
          </div>

          <button type="submit"
            className="w-full bg-green-600 hover:bg-green-700 text-white font-black text-3xl py-8 rounded-3xl shadow-2xl transition transform hover:scale-105">
            SHOW EXACT MARKET PRICE
          </button>
        </form>

        {result && (
          <div className="mt-16 text-center">
            <div className="bg-green-50 border-8 border-green-600 rounded-3xl p-12 inline-block">
              <p className="text-3xl font-black text-gray-800 mb-4">Your car is worth today:</p>
              <p className="text-7xl font-black text-green-600 mb-6">{result}</p>
              <p className="text-2xl text-gray-700">₦{result} (Lagos market price)</p>
              <a href="https://wa.me/2348123456789?text=I%20just%20valued%20my%20car%20on%20CarsAbeg!%20Can%20you%20beat%20this%20price?" 
                 className="mt-8 inline-block bg-yellow-400 text-black font-black text-2xl px-12 py-6 rounded-full hover:bg-yellow-300 transition">
                SELL IT TO US NOW →
              </a>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
